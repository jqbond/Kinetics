function temp(par)
    A = 2.75e8
    EA = par[1]
    a  = par[2]
    b  = par[3]
    rm(CA, CB, T) = A*exp(-EA/R/T)*CA^a*CB^b
    m1 = rm.(CA, CB[3], T[3])
    m2 = rm.(CA[3], CB, T[3])
    m3 = rm.(CA[3], CB[3], T)
    m  = vcat(m1, m2, m3)
    e1 = (d1 - m1)./d1
    e2 = (d2 - m2)./d2
    e3 = (d3 - m3)./d3
    e  = vcat(e1, e2, e3)
    SSE = sum(e.^2)
    return SSE, e, m
end

obj(par) = temp(par)[1]
